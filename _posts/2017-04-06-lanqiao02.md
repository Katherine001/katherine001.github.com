# 前言  
最近开始恶补数据结构基础，所以本次更新一道数据结构入门题。  

# 题目描述  

将一个单链表拆分为3个单链表

假设有一个带头节点的单链表L，每个节点值由单个数字、小写字母和大写字母构成。设计一个算法将其拆分成3个带头节点的单链表L1、L2和L3，L1包含L中的所有数字节点，L2包含L中的所有小写字母节点，L3包含L中的所有大写字母节点。

# 解题思路  

利用尾插法建3个单链表，遍历已知单链表，把数据域符合条件的节点筛选出来插入相应的单链表中即可。

# 代码实现  

```c
#include<stdio.h>
#include<stdlib.h>
 
#define MaxSize 30
 
typedef char ElemType;
typedef struct LNode{
    ElemType str;
    struct LNode *next;    
} LinkList;
 
void InitList(LinkList *&L)
{
    L = (LinkList *)malloc(sizeof(LinkList));
    L->next = NULL;
}
 
bool CreateList(LinkList *&L, int n)
{
    int i;
    LinkList *s;
     
    for(i = 0; i < n; i++)
    {
        s = (LinkList *)malloc(sizeof(LinkList));
        scanf("%c", &s->str);
        s->next = L->next;
        L->next = s;
    }
     
    return true;
}
 
void PrintList(LinkList *L)
{
    LinkList *p = L->next;
     
    while(p != NULL)
    {
        if(p->next == NULL)
        {
            printf("%c", p->str);
        }
        else
        {
            printf("%c->", p->str);
        } 
        p = p->next;
    }
    printf("\n");
}
 
void NumLinkList(LinkList *L, LinkList *&L1)
{
    LinkList *s, *p = L->next, *r = L1;
    while(p != NULL)
    {
        if(p->str >= '0' && p->str <= '9')
        {
            s = (LinkList *)malloc(sizeof(LinkList));
            s->str = p->str;
            r->next = s;
            r = s;
        }
        p = p->next;
    }
    s->next = NULL;
}
 
void UpLinkList(LinkList *L, LinkList *&L2)
{
    LinkList *s, *p = L->next, *r = L2;
    while(p != NULL)
    {
        if(p->str >= 'A' && p->str <= 'Z')
        {
            s = (LinkList *)malloc(sizeof(LinkList));
            s->str = p->str;
            r->next = s;
            r = s;
        }
        p = p->next;
    }
    s->next = NULL;
}
 
void LoLinkList(LinkList *L, LinkList *&L3)
{
    LinkList *s, *p = L->next, *r = L3;
    while(p != NULL)
    {
        if(p->str >= 'a' && p->str <= 'z')
        {
            s = (LinkList *)malloc(sizeof(LinkList));
            s->str = p->str;
            r->next = s;
            r = s;
        }
        p = p->next;
    }
    s->next = NULL;
}
 
int main()
{
    LinkList *L, *L1, *L2, *L3;
    int n, i;
    ElemType a[MaxSize];
    InitList(L);
    printf("Please input the numbers of nodes you want to create:");    
    scanf("%d", &n);
    getchar();
    printf("Please input the nodes's data:");
    if(CreateList(L, n))
    {
        printf("Input Successfully!\n");
    }
    printf("Linked List:");
    PrintList(L);
     
    InitList(L1);
    InitList(L2);
    InitList(L3);
     
    NumLinkList(L, L1);
    printf("Number Linked List:"); 
    PrintList(L1);
    UpLinkList(L, L2);
    printf("Uppercase Linked List:"); 
    PrintList(L2);
    LoLinkList(L, L3);
    printf("Lowercase Linked List:"); 
    PrintList(L3);
     
    return 0;
}

```
